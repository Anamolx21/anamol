from models import Chat

@app.post("/chat")
def chat(data: dict):
    db = SessionLocal()

    message = data.get("message")

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": "You are StudyLift AI."},
            {"role": "user", "content": message}
        ]
    )

    reply = response.choices[0].message.content

    # SAVE TO DATABASE
    chat_entry = Chat(
        user="admin",
        message=message,
        response=reply
    )

    db.add(chat_entry)
    db.commit()

    return {"reply": reply}
